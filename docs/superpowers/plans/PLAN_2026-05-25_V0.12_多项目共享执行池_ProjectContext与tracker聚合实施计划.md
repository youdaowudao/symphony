> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现只读 project-aware tracker 聚合，让候选 issue 带上 `ProjectContext`，但不接入真实 dispatch。

**Tech Stack:** Elixir / OTP、`SymphonyElixir.ProjectRegistry`、`SymphonyElixir.Tracker`、`SymphonyElixir.Linear.Client`、ExUnit、`SYMPHONY_TEST_MAX_CASES=4`。

  defmodule SymphonyElixir.ProjectContext do
    @moduledoc """
    Runtime project identity attached to tracker candidates.
    """

    @enforce_keys [:project_key, :display_name, :enabled, :max_concurrent_agents]
    defstruct [
      :project_key,
      :display_name,
      :enabled,
      :max_concurrent_agents
    ]

    @type t :: %__MODULE__{
            project_key: String.t(),
            display_name: String.t() | nil,
            enabled: boolean(),
            max_concurrent_agents: pos_integer()
          }

    @spec from_registry_entry(map()) :: {:ok, t()} | {:error, term()}
    def from_registry_entry(
          %{project_key: project_key, enabled: enabled, max_concurrent_agents: max_concurrent_agents} =
            entry
        )
        when is_binary(project_key) and is_boolean(enabled) and is_integer(max_concurrent_agents) and
               max_concurrent_agents > 0 do
      trimmed_key = String.trim(project_key)

      if trimmed_key == "" do
        {:error, {:invalid_project_context_entry, entry}}
      else
        display_name =
          entry
          |> Map.get(:display_name)
          |> normalize_display_name(trimmed_key)

        {:ok,
         %__MODULE__{
           project_key: trimmed_key,
           display_name: display_name,
           enabled: enabled,
           max_concurrent_agents: max_concurrent_agents
         }}
      end
    end

    def from_registry_entry(entry), do: {:error, {:invalid_project_context_entry, entry}}

    defp normalize_display_name(value, fallback) when is_binary(value) do
      case String.trim(value) do
        "" -> fallback
        display_name -> display_name
      end
    end

    defp normalize_display_name(_value, fallback), do: fallback
  end
  ```

  defmodule SymphonyElixir.Tracker.ProjectAggregation do
    @moduledoc """
    Builds project-aware tracker candidates without dispatching them.
    """

    alias SymphonyElixir.Linear.Issue
    alias SymphonyElixir.ProjectContext
    alias SymphonyElixir.Tracker.ProjectCandidate

    @type project_entry :: map()
    @type project_result :: %{
            project_key: String.t() | nil,
            status: :ok | :failed | :skipped,
            fetched_count: non_neg_integer(),
            candidate_count: non_neg_integer(),
            reason: term() | nil
          }
    @type aggregate_result :: %{
            candidates: [ProjectCandidate.t()],
            project_results: [project_result()]
          }

    @spec aggregate([project_entry()], (String.t() -> {:ok, [Issue.t()]} | {:error, term()})) ::
            {:ok, aggregate_result()}
            | {:error, {:all_project_fetches_failed, [project_result()]}}
            | {:error, {:invalid_project_entry, term()}}
    def aggregate(project_entries, fetcher) when is_list(project_entries) and is_function(fetcher, 1) do
      with {:ok, contexts} <- build_contexts(project_entries),
           {:ok, {candidates, project_results}} <- reduce_contexts(contexts, fetcher) do
        candidates = Enum.reverse(candidates)
        project_results = Enum.reverse(project_results)
        attempted_results = Enum.reject(project_results, &(&1.status == :skipped))

        if attempted_results != [] and Enum.all?(attempted_results, &(&1.status == :failed)) do
          {:error, {:all_project_fetches_failed, project_results}}
        else
          {:ok, %{candidates: candidates, project_results: project_results}}
        end
      end
    end

    defp build_contexts(project_entries) do
      case Enum.reduce_while(project_entries, {:ok, []}, fn entry, {:ok, contexts} ->
             case ProjectContext.from_registry_entry(entry) do
               {:ok, %ProjectContext{} = context} ->
                 {:cont, {:ok, [context | contexts]}}

               {:error, reason} ->
                 {:halt, {:error, {:invalid_project_entry, reason}}}
             end
           end) do
        {:ok, contexts} ->
          {:ok, Enum.reverse(contexts)}

        {:error, reason} ->
          {:error, reason}
      end
    end

    defp reduce_contexts(contexts, fetcher) do
      Enum.reduce(contexts, {:ok, {[], []}}, fn
        %ProjectContext{enabled: false} = context, {:ok, {acc_candidates, acc_results}} ->
          result = project_result(context.project_key, :skipped, 0, 0, nil)
          {:ok, {acc_candidates, [result | acc_results]}}

        %ProjectContext{} = context, {:ok, {acc_candidates, acc_results}} ->
          {:ok, fetch_project(context, fetcher, acc_candidates, acc_results)}
      end)
    end

    defp fetch_project(context, fetcher, acc_candidates, acc_results) do
      case fetcher.(context.project_key) do
        {:ok, issues} when is_list(issues) ->
          candidates = Enum.map(issues, &ProjectCandidate.new!(&1, context))
          result = project_result(context.project_key, :ok, length(issues), length(candidates), nil)
          {Enum.reverse(candidates, acc_candidates), [result | acc_results]}

        {:error, reason} ->
          result = project_result(context.project_key, :failed, 0, 0, reason)
          {acc_candidates, [result | acc_results]}
      end
    end

    defp project_result(project_key, status, fetched_count, candidate_count, reason) do
      %{
        project_key: project_key,
        status: status,
        fetched_count: fetched_count,
        candidate_count: candidate_count,
        reason: reason
      }
    end
  end
  ```
