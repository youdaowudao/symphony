# V0.22 Todo Pool 人工检查功能 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把首页 `Todo Pool` 占位区落成一次性、只读、人工触发的 Todo blocker 检查区，并保持它与 dispatch、runtime `blocked`、Linear 写入完全隔离。

**Architecture:** 新增独立的 `TodoPool.ManualCheck` 只读服务，直接读取 canonical project registry 和 Linear `Todo` issue 数据，返回一次性检查结果。`DashboardLive` 在点击 `手动检查` 后启动单次后台任务，完成后把结果保存在当前 LiveView assign 中渲染；结果不写回 orchestrator snapshot，也不进入 tracker dispatch 主链。UI 只展示 blocker 维度的检查结果，不新增任何状态修改动作。

**Tech Stack:** Elixir、Phoenix LiveView、`SymphonyElixir.ProjectRegistry`、`SymphonyElixir.Linear.Client`、ExUnit、`SYMPHONY_TEST_MAX_CASES=4`。

---

## 1. 计划边界

### 1.1 这次计划要做什么

- 新增一个与 orchestrator 解耦的只读手动检查服务，用来读取 enabled 项目的 `Todo` issue 和其前置 blocker 状态。
- 把 `DashboardLive` 的 `Todo Pool` 从占位文案改成真实只读结果展示。
- 在页面内增加“检查中 / 成功 / 项目级失败 / 全局失败”的最小状态机。
- 用 targeted tests 固定 `Done only` blocker 解除规则、`unknown` 规则、单项目失败隔离规则和“绝不触发写操作”的边界。

### 1.2 这次计划明确不做什么

- 不修改 `orchestrator` 的 dispatch 流程。
- 不修改 runtime `blocked` 的定义、分类或展示来源。
- 不新增 tracker callback，不把 Todo Pool 接进 `SymphonyElixir.Tracker` 的主接口。
- 不新增持久化 Todo 数据源、后台轮询或跨页面缓存。
- 不实现任何 Linear 写操作，包括 `update_issue_state/2` 和 `create_comment/2`。

```elixir
defmodule SymphonyElixir.TodoPool.ManualCheck do
  @moduledoc """
  One-shot, read-only Todo Pool inspection for the dashboard.
  """

  alias SymphonyElixir.{Linear.Client, ProjectRegistry}

  @todo_state "Todo"
  @issue_page_size 50
  @relation_page_size 20
  @todo_query """
  query SymphonyTodoPoolManualCheck($projectSlug: String!, $stateNames: [String!]!, $first: Int!, $relationFirst: Int!) {
    issues(filter: {project: {slugId: {eq: $projectSlug}}, state: {name: {in: $stateNames}}}, first: $first) {
      nodes {
        id
        identifier
        title
        state {
          name
        }
        inverseRelations(first: $relationFirst) {
          nodes {
            type
            issue {
              id
              identifier
              state {
                name
              }
            }
          }
        }
      }
    }
  }
  """

  def run(opts \\ []) do
    checked_at =
      Keyword.get(opts, :now, DateTime.utc_now())
      |> DateTime.truncate(:second)

    with {:ok, entries} <- project_registry_module(opts).normalized_entries() do
      projects =
        entries
        |> Enum.filter(&Map.get(&1, :enabled, false))
        |> Enum.map(&check_project(&1, linear_client_module(opts), checked_at))

      {:ok, %{checked_at: checked_at, projects: projects}}
    end
  end

  def classify_raw_issue_for_test(raw_issue), do: issue_payload(raw_issue)

  defp project_registry_module(opts),
    do: Keyword.get(opts, :project_registry_module, ProjectRegistry)

  defp linear_client_module(opts),
    do: Keyword.get(opts, :linear_client_module, Client)
end
```

defp check_project(entry, client_module, checked_at) do
  variables = %{
    projectSlug: entry.project_key,
    stateNames: [@todo_state],
    first: @issue_page_size,
    relationFirst: @relation_page_size
  }

  case client_module.graphql(@todo_query, variables) do
    {:ok, %{"data" => %{"issues" => %{"nodes" => nodes}}}} when is_list(nodes) ->
      issue_payloads = Enum.map(nodes, &issue_payload/1)

      %{
        checked_at: checked_at,
        project_key: entry.project_key,
        project_display_name: entry.display_name || entry.project_key,
        todo_count: length(issue_payloads),
        ready_count: Enum.count(issue_payloads, &(&1.status == "ready")),
        blocked_count: Enum.count(issue_payloads, &(&1.status == "blocked_by_dependency")),
        unknown_count: Enum.count(issue_payloads, &(&1.status == "unknown")),
        issues: issue_payloads,
        project_error: nil
      }

    {:ok, unexpected} ->
      %{
        checked_at: checked_at,
        project_key: entry.project_key,
        project_display_name: entry.display_name || entry.project_key,
        todo_count: nil,
        ready_count: nil,
        blocked_count: nil,
        unknown_count: nil,
        issues: [],
        project_error: {:invalid_todo_pool_response, unexpected}
      }

    {:error, reason} ->
      %{
        checked_at: checked_at,
        project_key: entry.project_key,
        project_display_name: entry.display_name || entry.project_key,
        todo_count: nil,
        ready_count: nil,
        blocked_count: nil,
        unknown_count: nil,
        issues: [],
        project_error: reason
      }
  end
end

defp issue_payload(%{"inverseRelations" => %{"nodes" => relations}} = issue) when is_list(relations) do

    title: issue["title"],
    state: get_in(issue, ["state", "name"]),
    status: blocker_status(relations, blockers),

end

    issue_identifier: issue["identifier"],
    title: issue["title"],

- [ ] **Step 3: 实现 `Done only` blocker 判定，禁止复用 terminal state 集合**

```elixir
defp blocker_status(relations, blockers) do
  block_relations = Enum.filter(relations, &blocks_relation?/1)

  cond do
    Enum.any?(block_relations, &blocker_relation_unknown?/1) -> "unknown"
    block_relations == [] -> "ready"
    Enum.any?(blockers, &(blocker_done_state?(&1.blocker_state) == false)) -> "blocked_by_dependency"
    Enum.all?(blockers, &blocker_done_state?(&1.blocker_state)) -> "ready"
    true -> "unknown"
  end
end

defp extract_blockers(relations) do
  relations
  |> Enum.filter(&blocks_relation?/1)
  |> Enum.map(&blocker_payload/1)
end

defp blocks_relation?(%{"type" => type}) when is_binary(type),
  do: String.downcase(String.trim(type)) == "blocks"

defp blocks_relation?(_relation), do: false

defp blocker_relation_unknown?(%{"issue" => %{"id" => id, "identifier" => identifier, "state" => %{"name" => state}}})
     when is_binary(id) and is_binary(identifier) and is_binary(state),
     do: String.trim(state) == ""

defp blocker_relation_unknown?(_relation), do: true

defp blocker_payload(relation) do
  %{
    blocker_id: get_in(relation, ["issue", "id"]),
    blocker_identifier: get_in(relation, ["issue", "identifier"]),
    blocker_state: get_in(relation, ["issue", "state", "name"])
  }
end

defp blocker_done_state?(state) when is_binary(state),
  do: String.downcase(String.trim(state)) == "done"

defp blocker_done_state?(_state), do: false
```

def handle_event("manual_check_todo_pool", _params, socket) do
  parent = self()
  checker = todo_pool_manual_check_module()
  requested_at = DateTime.utc_now() |> DateTime.truncate(:second)

  Task.start(fn ->
    result =
      try do
        checker.run(notify: parent)
      rescue
        exception ->
          {:error, {:todo_pool_manual_check_exception, Exception.message(exception)}}
      catch
        kind, reason ->
          {:error, {:todo_pool_manual_check_throw, {kind, reason}}}
      end

    send(parent, {:todo_pool_manual_check_finished, result})
  end)

  {:noreply,
   assign(socket, :todo_pool_check, %{
     status: :checking,
     requested_at: requested_at,
     checked_at: nil,
     projects: [],
     error: nil
   })}
end
